select count(Persons.Name) From Persons
join PersonsStudySubjects on Persons.Id = PersonsStudySubjects.StudentId
join StudySubjects on StudySubjects.Id = PersonsStudySubjects.SubjectId
join Universities on Universities.Id = StudySubjects.UniversityId
where Persons.Age > 30
group by Universities.Id
